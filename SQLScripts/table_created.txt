  CREATE TABLE HR.EMPLOYEE_LIST 
   (	USERID NUMBER(5) NOT NULL, 
	USERNAME VARCHAR2(10), 
	USERKEY VARCHAR2(8), 
	FNAME VARCHAR2(15), 
	LNAME VARCHAR2(20), 
	POSITION VARCHAR2(8)
   );

  CREATE UNIQUE INDEX HR.EMPLOYEE_LIST_PK ON HR.EMPLOYEE_LIST ("USERID");
  
  ALTER TABLE HR.EMPLOYEE_LIST ADD CONSTRAINT EMPLOYEE_LIST_PK PRIMARY KEY ("USERID");
  
  CREATE SEQUENCE EMPLOYEE_SEQ
  START WITH 1
  INCREMENT BY 1; 

  CREATE OR REPLACE TRIGGER HR.TRIGGER1
  BEFORE INSERT ON HR.EMPLOYEE_LIST
  FOR EACH ROW
  BEGIN
  SELECT EMPLOYEE_SEQ.NEXTVAL INTO :NEW.USERID FROM DUAL;
  END;

  ALTER TRIGGER HR.TRIGGER1 ENABLE;








  CREATE TABLE HR.PROJECT 
   (	USERID NUMBER(5) NOT NULL, 
	PROJECTID NUMBER(8) NOT NULL, 
	PROJECTNAME VARCHAR2(20)
   );

  CREATE UNIQUE INDEX HR.PROJECT_PK ON HR.PROJECT ("PROJECTID");

  ALTER TABLE HR.PROJECT ADD CONSTRAINT PROJECT_PK PRIMARY KEY ("PROJECTID");

  ALTER TABLE HR.PROJECT ADD CONSTRAINT PROJECT_FK1 FOREIGN KEY ("USERID")
	  REFERENCES HR.EMPLOYEE_LIST ("USERID") ENABLE;

  CREATE SEQUENCE PROJECT_SEQ

  INCREMENT BY 1;

  ALTER TRIGGER HR.TRIGGER2 ENABLE;
